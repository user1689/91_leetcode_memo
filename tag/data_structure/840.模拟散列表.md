## python3
```python3
N = 100003
h = [-1] * N
e = [0] * N 
ne = [0] * N
idx = 0

def insert(x):
    global idx, N
    k = (x % N + N) % N
    # k = x % N 
    e[idx] = x
    ne[idx] = h[k]
    h[k] = idx
    idx += 1
    
def find(x):
    global idx, N
    k = (x % N + N) % N
    # k = x % N 
    i = h[k]
    while (i != -1):
        if(e[i] == x):
            return True
        i = ne[i]
    return False
    
def main():
    global idx, N
    n = int(input())
    for i in range(n):
        opt, x = list(input().split(" "))
        if (opt == "I"):
            insert(int(x))
        elif (opt == "Q"):
            flag = find(int(x))
            if (flag):
                print("Yes")
            else:
                print("No")

main()

```
