```python3
## 拉链法
N = 100003
h = [-1] * N
e = [0] * N 
ne = [0] * N
idx = 0

def insert(x):
    global idx, N
    k = (x % N + N) % N
    # k = x % N 
    e[idx] = x
    ne[idx] = h[k]
    h[k] = idx
    idx += 1
    
def find(x):
    global idx, N
    k = (x % N + N) % N
    # k = x % N 
    i = h[k]
    while (i != -1):
        if(e[i] == x):
            return True
        i = ne[i]
    return False
    
def main():
    global idx, N
    n = int(input())
    for i in range(n):
        opt, x = list(input().split(" "))
        if (opt == "I"):
            insert(int(x))
        elif (opt == "Q"):
            flag = find(int(x))
            if (flag):
                print("Yes")
            else:
                print("No")

main()


## 开放寻址法
N = 200003
h = [0x3f3f3f3f] * N
'''
    首先，0x3f3f3f的体验感很好，0x3f3f3f3f的十进制是1061109567，也就是10^9级别的
    （和0x7fffffff一个数量级），而一般场合下的数据都是小于10^9的，所以它可以作为无穷大
    使用而不致出现数据大于无穷大的情形。
    比如0x3f3f3f3f+0x3f3f3f3f=2122219134，这非常大但却没有超过32-bit，int的表示范围，
    所以0x3f3f3f3f还满足了我们“无穷大加无穷大还是无穷大”的需求。
    但是INF不同，一旦加上某个值，很容易上溢，数值有可能转成负数，有兴趣的小伙伴可以去试一试
'''

def find(x):
    global N
    k = (x % N + N) % N
    while (h[k] != 0x3f3f3f3f and h[k] != x):
        k += 1
        if (k == N):
            k = 0 
    return k

def main():
    n = int(input())
    for i in range(n):
        opt, x = list(input().split(" "))
        x = int(x)
        k = find(x)
        if (opt == "I"):
            h[k] = x
        elif (opt == "Q"):
            if (h[k] == x):
                print("Yes")
            else:
                print("No")
main()
```
