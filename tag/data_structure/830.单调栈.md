```python3
N = 100010
stack = [0] * N
tt = -1

def main():
    global tt
    input()
    s = list(map(int, input().rstrip().split(" ")))
    
    for i in range(len(s)):
        x = s[i]
        while (tt != -1 and stack[tt] >= x):
            tt -= 1
        if (tt != -1):
            print(stack[tt], end = " ")
        else:
            print("-1", end = " ")
        tt += 1
        stack[tt] = x

main()

```
