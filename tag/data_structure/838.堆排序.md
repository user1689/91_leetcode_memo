## 题目
https://www.acwing.com/problem/content/description/840/

## python3
```python3
size = 0
N = 100010
h = [0] * N

def down(u):
    global size, h
    t = u
    l, r = u * 2, u * 2 + 1
    if (l <= size and h[l] < h[t]):
        t = l
    if (r <= size and h[r] < h[t]):
        t = r
    if (t != u):
        h[u], h[t] = h[t], h[u]
        down(t)

def up(u):
    while (u // 2 > 0 and h[u // 2] > h[u]):
        h[u // 2], h[u] = h[u], h[u // 2]
        u = u // 2
    

def main():
    global size, h
    n, m = list(map(int, input().split(" ")))
    h = list(map(int, input().split(" ")))
    h = [0] + h
    size = n
    
    i = n // 2
    while (i > 0):
        down(i)
        i -= 1
    i = 1
    
    while (m > 0):
        m -= 1
        print(h[1], end=" ")
        
        tmp = h[1]
        h[1] = h[size]
        h[size] = tmp
        size -= 1
        
        down(1)
        up(1)
main()
```
