def qmi(a, b, p):
    res = 1
    while(b):
        if (b & 1):
            res = res * a % p
        a = a * a % p
        b >>= 1
    return res


def main():
    n = int(input())
    for i in range(n):
        a, b, p = map(int, input().split(" "))
        res = qmi(a, b, p)
        print(res)
    
main()
