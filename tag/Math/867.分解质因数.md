def main():
    n = int(input())
    for i in range(n):
        x = int(input())
        j = 2
        while (j <= x // j):
            if (x % j == 0):
                s = 0
                while (x % j == 0):
                    x //= j
                    s += 1
                print(j, end=" ")
                print(s)
            j += 1

        
        if (x > 1):
            print(x, end=" ")
            print(1)
        
        print(" ")
main()
