N = 100010
fact = [0 for _ in range(N)]
infact = [0 for _ in range(N)]
fact[0] = 1
infact[0] = 1

def qmi(a, b, p):
    res = 1
    while (b):
        if (b & 1):
            res = res * a % p
        a = a * a % p
        b >>= 1
    return res


def main():
    global N
    p = int(1e9 + 7)
    n = int(input())
    for i in range(1, N):
        fact[i] = fact[i - 1] * i % p
        infact[i] = infact[i - 1] * qmi(i, p - 2, p) % p
        
    for i in range(n):
        a, b = map(int, input().split())
        res = fact[a] * infact[a - b] % p * infact[b] % p
        print(res)

main()
