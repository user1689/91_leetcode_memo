https://www.acwing.com/problem/content/892/

## python3
```python3
def main():
    res = 0
    M = 20
    p = [0] * M
    n, m = map(int, input().split())
    tmp = list(map(int, input().split()))
    for i in range(0, len(tmp)):
        p[i] = tmp[i]
    
    def dfs(idx, times, opt):
        nonlocal res, n, m
        for i in range(idx, m):
            new_times = times * p[i] 
            res += opt * (n // new_times)
            dfs(i + 1, new_times, -opt)    
    
    dfs(0, 1, 1)

    print(res)

main()
        
```
