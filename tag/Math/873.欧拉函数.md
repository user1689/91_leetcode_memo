def main():
    n = int(input())
    for _ in range(n):
        x = int(input())
        
        res = x
        i= 2
        while ( i <= x // i ):
            if (x % i == 0):
                # res = res * (1 - (1 // i))
                # 代码中，因为在公式中(1 − 1/a)很容易出现小数，这是不允许的，所以我们可以变化一下：
                # (1 - 1/a) = (a/a − 1/a) = a−1/a

                # 所以在代码中我们可以先除以a再乘上a−1
                
                res = res // i * (i - 1)
                while (x % i == 0):
                    x //= i
            i += 1
        if (x > 1):
            res =  res // x * (x - 1)
    
        print(res)

main()
