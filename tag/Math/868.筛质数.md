## 题目
https://www.acwing.com/problem/content/description/870/

## 思路
```python3
# 最普通的筛法 O(nlogn)
N = 1000010
st = [False] * N
primes = [0] * N
cnt = 0

def main():
    global cnt
    x = int(input())
    for i in range(2, x+1):
        # 把素数存起来
        if (not st[i]):
            primes[cnt] = i
            cnt += 1
        # 不管是合数还是质数，都用来筛掉后面它的倍数
        for j in range(i, x+1, i):
            st[j] = True
    print(cnt)
    
main()

# 诶氏筛法 O(nloglogn)
N = 1000010
st = [False] * N
primes = [0] * N
cnt = 0

def main():
    global cnt
    x = int(input())
    
    for i in range(2, x+1):
        if (not st[i]):
            primes[cnt] = i
            cnt += 1
            # 可以用质数就把所有的合数都筛掉；
            for j in range(i, x+1, i):
                st[j] = True
    print(cnt)
    
main()

# 线性筛 O(n)
N = 1000010
cnt = 0
st = [False] * N
primes = [0] * N

def main():
    global cnt 
    x = int(input())
    for i in range(2, x + 1):
        if (not st[i]):
            primes[cnt] = i
            cnt += 1
        j = 0
        while (primes[j] <= x / i):
            st[primes[j] * i] = True
            if (i % primes[j] == 0):
                break
            j += 1
    print(cnt)
    
main()
```
