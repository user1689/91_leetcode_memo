题目: https://www.acwing.com/problem/content/description/829/

```python3
N = 100010
e = [0] * N
l = [0] * N
r = [0] * N
idx = 0

    
# 初始化01
def init():
    global idx
    # 0表示左端点 1表示右端点
    r[0] = 1
    l[1] = 0
    idx = 2
    
# 初始化02
# r[0], l[1], idx = 1, 0, 2


# 在下标是k的点的右边 插入一个x
def insert(k, x):
    global idx;
    e[idx] = x 
    r[idx] = r[k]
    l[idx] = k 
    l[r[k]] = idx # 注意顺序 先更改r[k]的左指针 再去更改r[k]
    r[k] = idx
    # l[r[idx]] = idx
    idx += 1
    
# 删除第k个点
def remove(k):
    r[l[k]] = r[k]
    l[r[k]] = l[k]

def main():
    global idx
    init()
    m = int(input())
    while (m > 0):
        s = list(input().split(" "))
        opt = s[0]
        if (opt == "L"):
            x = int(s[1])
            insert(0, x)
        elif (opt == "R"):
            x = int(s[1])
            insert(l[1], x) #
        elif (opt == "D"):
            k = int(s[1])
            remove(k + 1)
        elif (opt == "IL"):
            k = int(s[1])
            x = int(s[2])
            insert(l[k + 1], x)
        else:
            k = int(s[1])
            x = int(s[2])
            insert(k + 1, x)
        m -= 1
    i = r[0]
    while (i != 1):
        print(e[i], end=" ")
        i = r[i]
main()
```
