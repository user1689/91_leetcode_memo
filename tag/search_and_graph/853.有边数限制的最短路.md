```python3

N = 510
M = 10010
dist = [0x3f3f3f3f] * N
e = [[-1, -1, -1] for _ in range(M)]
n = 0
m = 0
k = 0

def bell_ford():
    global n, m, k
    dist[1] = 0
    # iterate k times
    for i in range(k):
    '''
    注意：back[] 数组是上一次迭代后 dist[] 数组的备份，由于是每个点同时向外出发，
    因此需要对 dist[] 数组进行备份，若不进行备份会因此发生串联效应，影响到下一个点
    '''
        back = dist.copy()
        for j in range(0, m):   
            a = e[j][0]
            b = e[j][1]
            w = e[j][2]
            dist[b] = min(dist[b], back[a] + w)

    return dist[n]

def main():
    global N, M, n, m, k
    n, m, k = map(int, input().split())
    for i in range(m):
        x, y, z = map(int, input().split())
        e[i][0] = x
        e[i][1] = y
        e[i][2] = z
    
    res = bell_ford()
    '''
    是否能到达n号点的判断中需要进行if(dist[n] > 0x3f3f3f3f / 2)判断，
    而并非是if(dist[n] == 0x3f3f3f3f)判断，原因是INF是一个确定的值，
    并非真正的无穷大，会随着其他数值而受到影响，dist[n]大于某个与0x3f3f3f3f相同数量级的数即可
    '''
    if (res > 0x3f3f3f3f // 2):
        print("impossible")
    else:
        print(res)
    return 0
    
main()
```
