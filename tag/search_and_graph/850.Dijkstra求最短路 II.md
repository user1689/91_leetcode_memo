https://www.acwing.com/problem/content/description/852/

## python3
```python3
import collections, heapq

N = 150010
dist = [0x3f3f3f3f] * N
st = [False] * N

# 稀疏图采用邻接表
h = [-1] * N
e = [0] * N
ne = [0] * N 
idx = 0
w = [0] * N # 存储边的权重

n, m = 0, 0

def add(a, b, c):
    global idx
    e[idx] = b
    w[idx] = c
    ne[idx] = h[a]
    h[a] = idx
    idx += 1

def dijkstra():
    global n 
    heap = []
    dist[1] = 0
    heapq.heappush(heap, [0, 1])
    
    while(heap):
        # 弹出距离起点距离最近的点
        distance, t = heapq.heappop(heap)
        
        # 原本需要for遍历n次 然后再for每个点对比出距离起点最小值 并且用最小值更新其他点到起点的距离 O(n^2)
        # 用没有被访问过 并且距离起点距离最近的点去更新其他距离 
        if (st[t]):
            continue
        st[t] = True
        i = h[t]
        while(i != -1):
            j = e[i]
            if (dist[j] > distance + w[i]):
                dist[j] = distance + w[i]
                heapq.heappush(heap, [dist[j], j])
            i = ne[i]
    if (dist[n] == 0x3f3f3f3f):
        return -1
    return dist[n]
        
def main():
    global n
    n, m = map(int, input().split())
    for i in range(m):
        x, y, z = map(int, input().split())
        add(x, y, z)
    
    print(dijkstra())
    return 0

main()

```
